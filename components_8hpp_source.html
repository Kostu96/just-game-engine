44
45 glm::mat4 getTransform() const
46 {
47 glm::mat4 rotMatrix = glm::rotate(glm::mat4{ 1.f }, glm::radians(rotation.x), { 1.f, 0.f, 0.f }) *
48 glm::rotate(glm::mat4{ 1.f }, glm::radians(rotation.y), { 0.f, 1.f, 0.f }) *
49 glm::rotate(glm::mat4{ 1.f }, glm::radians(rotation.z), { 0.f, 0.f, 1.f });
50
51 return glm::translate(glm::mat4{ 1.f }, translation) * rotMatrix * glm::scale(glm::mat4{ 1.f }, scale);
52 }
