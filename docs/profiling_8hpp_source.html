67 template<size_t SIZE>
68 constexpr auto cleanupStringForProfiler(const char(&str)[SIZE])
69 {
70 constexpr size_t arrSize = 3;
71 constexpr const char* remove[arrSize] = {
72 "__cdecl ",
73 "class ",
74 "struct "
75 };
76 constexpr size_t removeSize[3] = { 9, 7, 8 };
77 char result[SIZE];
78
79 size_t dstIndex = 0;
80 for (size_t srcIndex = 0; srcIndex < SIZE; ++srcIndex)
81 {
82 for (size_t i = 0; i < arrSize; ++i)
83 {
84 size_t matchIndex = 0;
85 while (matchIndex < removeSize[i] - 1 && srcIndex + matchIndex < SIZE - 1 && str[srcIndex + matchIndex] == remove[i][matchIndex])
86 matchIndex++;
87 if (matchIndex == removeSize[i] - 1)
88 srcIndex += matchIndex;
89 }
90
91 result[dstIndex++] = str[srcIndex] == '"' ? '\'' : str[srcIndex];
92 }
93 return std::string(result);
94}
