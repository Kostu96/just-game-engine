82 glm::rotate(glm::mat4{ 1.f }, glm::radians(rotation.y), { 0.f, 1.f, 0.f }) *

85 return glm::translate(glm::mat4{ 1.f }, translation) * rotMatrix * glm::scale(glm::mat4{ 1.f }, scale);
